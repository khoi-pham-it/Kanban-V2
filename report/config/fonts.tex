% File này chứa cấu hình về font chữ và ngôn ngữ tiếng Việt

% --- Cấu hình ngôn ngữ ---
\usepackage[vietnamese]{babel}
\usepackage{fontspec}

% --- Chọn font chữ ---
% Sử dụng Extension=.ttf để tránh lỗi font với XeLaTeX
\setmainfont{CrimsonPro}[
    Extension=.ttf,
    UprightFont=*-Regular,
    BoldFont=*-Bold,
    ItalicFont=*-Italic,
    BoldItalicFont=*-BoldItalic,
    Path=/home/kiet/.local/share/fonts/Crimson_Pro/static/
]

\setsansfont{OpenSans}[
    Extension=.ttf,
    UprightFont=*-Regular,
    BoldFont=*-Bold,
    ItalicFont=*-Italic,
    BoldItalicFont=*-BoldItalic,
    Path=/home/kiet/.local/share/fonts/Open_Sans/static/
]

% Sử dụng JetBrains Mono cho các khối code (tốt cho tiếng Việt)
\setmonofont{JetBrainsMono-Regular.ttf}[
    Scale=0.9,
    Path=/usr/share/fonts/TTF/,
    BoldFont=JetBrainsMono-Bold.ttf,
    ItalicFont=JetBrainsMono-Italic.ttf,
    BoldItalicFont=JetBrainsMono-BoldItalic.ttf,
    RawFeature={+ccmp,+locl,+mark,+mkmk}
]

% --- Điều chỉnh kích thước font ---
\usepackage{anyfontsize}

\renewcommand{\normalsize}{\fontsize{13}{20}\selectfont}
\renewcommand{\small}{\fontsize{11}{13}\selectfont}
\renewcommand{\footnotesize}{\fontsize{10}{12}\selectfont}
\renewcommand{\large}{\fontsize{14}{17}\selectfont}
\renewcommand{\Large}{\fontsize{16}{19}\selectfont}
\renewcommand{\LARGE}{\fontsize{18}{22}\selectfont}
\renewcommand{\huge}{\fontsize{20}{24}\selectfont}
\renewcommand{\Huge}{\fontsize{24}{28}\selectfont}
