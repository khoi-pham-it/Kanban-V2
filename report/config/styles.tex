% File này chứa các macro tùy chỉnh và định nghĩa style cho dự án

% --- CẤU HÌNH XỬ LÝ TRÀN LỀ (OVERFULL HBOX) ---
\setlength{\emergencystretch}{3em} % Cung cấp thêm không gian co giãn để tránh từ dài bị tràn lề
\tolerance=9999 % Chấp nhận các dòng có khoảng trắng lớn hơn bình thường
\hbadness=9999 % Ẩn các cảnh báo underfull hbox không cần thiết

% --- TRANG TRÍ TIÊU ĐỀ CHƯƠNG ---
\titleformat{\chapter}[display]
  {\normalfont\bfseries\color{black}}
  {\Large\chaptertitlename\ \thechapter}
  {1ex}
  {\huge\MakeUppercase}
  [\vspace{1ex}\rule{\textwidth}{1pt}]

% Điều chỉnh khoảng cách trên và dưới của chapter
\titlespacing{\chapter}{0pt}{0pt}{3ex}

% --- TRANG TRÍ TIÊU ĐỀ PHẦN (PART) ---
\titleformat{\part}[display]
  {\centering\normalfont\huge\bfseries}
  {\partname\ \thepart}
  {2ex}
  {\Huge\MakeUppercase}
