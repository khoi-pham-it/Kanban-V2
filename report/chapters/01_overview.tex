\chapter{Giới thiệu về ReactJS}

\section{Lịch sử và Bối cảnh}
\begin{itemize}
    \item Được phát triển bởi Facebook (Meta) và ra mắt chính thức vào năm 2013.
    \item \textbf{Vấn đề cốt lõi:} React sinh ra để giải quyết vấn đề hiệu năng và quản lý dữ liệu khi ứng dụng web trở nên cực kỳ phức tạp và có nhiều trạng thái thay đổi liên tục.
\end{itemize}

\section{Kiến trúc và Nguyên lý hoạt động}
\begin{itemize}
    \item \textbf{Virtual DOM:} Tối ưu hóa việc cập nhật giao diện thông qua quá trình diffing.
    \item \textbf{Component-based architecture:} Ứng dụng được chia nhỏ thành các thành phần độc lập (App > Router > AuthProvider > Layout...).
    \item \textbf{One-way Data Flow:} Dữ liệu chảy từ cha xuống con, giúp kiểm soát trạng thái dễ dàng.
\end{itemize}

\section{So sánh và Đánh giá (Critical Thinking)}
\begin{itemize}
    \item \textbf{So sánh:} React vs Vue/Angular. React có tính linh hoạt cao hơn nhờ kiến trúc mở và thư viện phong phú, nhưng yêu cầu người lập trình phải tự đưa ra nhiều quyết định cấu trúc (Design Patterns).
    \item \textbf{Phân tích Trade-off:} NÊN dùng cho các Single Page App (SPA) cần tương tác phức tạp (như bảng Kanban). KHÔNG NÊN dùng cho các website tĩnh cực đơn giản chỉ cần vài trang HTML thuần.
\end{itemize}
